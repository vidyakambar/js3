7.Check whether a Number is a prime or not.

const Prime_Check = (N) => 
        {
            if(N===1)
            {
                return "NO";
            }
            else
            {
                for(let i=2; i<=Math.sqrt(N); ++i)
                {
                    if(N%i===0)
                    {
                        return "NO";
                    }
                }
                return "YES";
            }
        };
